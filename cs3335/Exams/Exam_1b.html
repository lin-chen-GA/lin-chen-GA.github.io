    • Question 1
  • An Armstrong number of three digits is an integer such that the sum of the cubes of its digits is equal to the number itself. For example, 371 is an Armstrong number since 3^3 + 7^3 + 1^3 = 371
  • Write a program that reads a three-digit integer and finds out whether it is an Armstrong number or not. Run the program and enter different positive integer numbers, you should have the following outputs
    • Enter a positive integer number: 23
23 is not a three-digit number

Enter a positive integer number: 371
371 is a Armstrong number

Enter a positive integer number: 235
235 is not a Armstrong number
		
    /* main.c
 */
#include <stdio.h>
#include "util.h"

int main()
{
	int num = read_num();

	if(isLegal(num))
	{
		if(isArmstrong(num))
		{
			printf("%d is a Armstrong number ...\n", num);
		}
		else
		{
			printf("%d is not a Armstrong number ...\n", num);
		}
	}
	else
		printf("%d is not a three-digit number ...\n", num);

	return 0;
}

#endif
