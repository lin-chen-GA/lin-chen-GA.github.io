    // main.c
#include <stdio.h>
#include "io.h"
#include "array.h"

int main()
{
	int size = getNum();

	int *array = getArray(size);

	display(array, size);

	sortArray(array, size);

	display(array, size);

	int *rd = removeDuplicates(array, size);

	display(rd, size);

	free(array);
	free(rd);

	return 0;
}
