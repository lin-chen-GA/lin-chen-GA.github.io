// util.c
#include <stdio.h>
#include <stdlib.h>
#include <setjmp.h>
#include "util.h"

extern jmp_buf buf;

int *getArray(int size)
{
	int *ptr = (int *) malloc(size*sizeof(int));

	for(int i = 0; i < size; i++)
		ptr[i] = i;

	return ptr;
}

// array, pointer

// index, element index

{
