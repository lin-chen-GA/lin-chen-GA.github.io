#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include "io.h"
#include "util.h"

int main()
{
	srand(time(NULL));

	printf("Read row number: ");
	int row = readNum();
	printf("Read column number: ");
	int column = readNum();

	int array[row][column];
	assign2DArray(row, column, array);
	display2DArray(row, column, array);

	int array2[row][column];
	assign2DArray(row, column, array2);
	display2DArray(row, column, array2);

	int sum[row][column];
	sum2DArrays(row, column, array, array2, sum);

	display2DArray(row, column, sum);

	return 0;
}

# io.c
#include <stdio.h>
#include "io.h"

int readNum()
{
	int num;
	scanf("%d", &num);

	return num;
}

void display2DArray(int row, int column, int array[][column])
{
	for(int i = 0; i < row; i++)
	{
		for(int j = 0; j < column; j++)
			printf("%5d", array[i][j]);
		printf("\n");
	}
	printf("\n");
}
		
// util.h
#ifndef UTIL_H
#define UTIL_H

void assign2DArray(int row, int column, int array[][column]);

void sum2DArrays(int row, int column, int array[][column], int array2[][column], int sum[][column]);

#endif
		
// util.c
#include <stdio.h>
#include <stdlib.h>
#include "util.h"

void assign2DArray(int row, int column, int array[][column])
{
	for(int i = 0; i < row; i++)
		for(int j = 0; j < column; j++)
			array[i][j] = rand()%10+1;
}

void sum2DArrays(int row, int column, int array[][column], int array2[][column], int sum[][column])
{
	for(int i = 0; i < row; i++)
		for(int j = 0; j < column; j++)
			sum[i][j] = array[i][j] + array2[i][j];
}
