#include <ctype.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

char *getString()
{
	printf("Enter a string:\n");
	char *str = (char *) malloc(100*sizeof(char));
	fgets(str, 100, stdin);

	return str;
}

char *toUpper(char *str)
{
	int size = strlen(str);

	char *upper = (char *)malloc((size+1)*sizeof(char));

	for(int i = 0; i <= size; i++)
		upper[i] = toupper(str[i]);

	upper[size-1] = '\0'; // remove \n character

	return upper;
}

int main()
{
	char *str = getString();

	char *strUpper = toUpper(str);

	printf("%s\n", strUpper);

	return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdbool.h>

char *getString()
{
	char *str = (char *) malloc(100*sizeof(char));
	fgets(str, 100, stdin);

	str[strlen(str)-1] = '\0'; // remove \n character

	return str;
}

bool isPalindrome(char *s1, char *s2)
{
	if(strcmp(s1, s2) == 0)
		return true;
	return false;
}

char *reverse(char *s)
{
	int size = strlen(s);

	char *ptr = (char *)malloc((size+1)*sizeof(char));

	for(int i = size-1; i >= 0; i--)
		ptr[size-1-i] = s[i];

	ptr[size] = '\0';

	return  ptr;
}

int main()
{
	char *s1 = getString();
	char *s2 = reverse(s1);

	if(isPalindrome(s1, s2))
		printf("%s is palindrome\n", s1);
	else
		printf("%s is not a palindrome\n", s1);

	free(s1);
	free(s2);

	return 0;
}

#include <stdio.h>
#include <stdlib.h>

int getInteger()
{
	printf("Enter an integer number:\n");
	char str[100];
	scanf("%s", str); // read the first token

	return atoi(str);
}

int getSum(int num)
{
	int s = 0;

	for(int i = 1; i <= num; i++)
		s += i;

	return s;
}

int main()
{
	int num = getInteger();

	printf("Sum of 1 to %d is %d\n", num, getSum(num));

	return 0;
}
