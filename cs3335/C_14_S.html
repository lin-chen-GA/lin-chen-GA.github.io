#include <stdio.h>

int * getMax(int *n_1, int *n_2)
{
	if(*n_1 >= *n_2)
		return n_1;
	else
		return n_2;
}

int main()
{
	printf("Enter two integer numbers: ");
	int num_1, num_2;
	scanf("%d", &num_1);
	scanf("%d", &num_2);

	int *ptr;
	ptr = getMax(&num_1, &num_2);

	printf("Max: %d\n", *ptr);

	return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

int * getNums(int size)
{
	int *ptr = NULL;
	ptr = (int *) malloc(size*sizeof(int));
	printf("Please enter %d numbers: \n", size);
	int num;

	for(int i = 0; i < size; i++)
	{
		scanf("%d", &num);
		*(ptr+i) = num;
	}

	return ptr;
}

void display(int *ptr, int size)
{
	for(int i = 0; i < size; i++)
		printf("%5d", *(ptr+i));
	printf("\n");
}

int getGCD(int *ptr, int size)
{
	// search minimum element
	// assume elements are positive integers
	int m = ptr[0];

	for(int i = 1; i < size; i++)
		if(ptr[i] > m)
			m = ptr[i];

	int gcd = 1;
	for(int i = 1; i <= m; i++)
	{
		bool divided = true;
		for(int j = 0; j < size; j++)
		{
			if (ptr[j] % i != 0)
			{
				divided = false;
				break;
			}
		}
		if(divided)
			gcd = i;
	}

	return gcd;
}

int main()
{
	int *p = getNums(5);

	display(p, 5);

	printf("GCD: %d\n", getGCD(p, 5));

	return 0;
}
