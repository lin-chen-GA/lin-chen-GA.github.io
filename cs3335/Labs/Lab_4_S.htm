// io.c
#include <stdio.h>
#include "io.h"

int read_year()
{
	int year;

	printf("Eneter a year: ");
	scanf("%d", &year);

	return year;
}
		
// util.c
#include <stdbool.h>

bool is_leap_year(int year)
{
	if(year % 4 == 0)
	{
		if(year % 100 == 0)
		{
			if(year % 400 == 0)
				return true;
			else
				return false;
		}
		else
			return true;
	}
	else
		return false;
}
