#include <stdio.h>
#include <stdlib.h>
#include <time.h>
 
int main()
{
		srand(time(NULL));
		 
		int dice_1, dice_2;
		int sum = 0;
		float average;

		for(int i = 0; i < 1000000; i++)
		{
			dice_1 = rand()%6 + 1;
			dice_2 = rand()%6 + 1;

			sum += dice_1 + dice_2;

			printf("%10dth:%10d%10d\n", i+1, dice_1, dice_2);
		}

		average = sum / 1000000.0;

		printf("Average Number: %10.2f\n", average);
			 
		return 0;
}
