#include <stdio.h>

int fib_r(int n)
{
	if(n == 0 || n == 1)
		return n;
	return fib_r(n-1) + fib_r(n-2);
}

int fib_s(int n)
{
	if(n == 0 || n == 1)
		return n;

	int array[n+1];
	array[0] = 0;
	array[1] = 1;

	for(int i = 2; i <= n; i++)
		array[i] = array[i-1] + array[i-2];

	return array[n];
}

int getNum()
{
	int num;

	printf("Enter an positive integer: ");
	scanf("%d", &num);

	return num;
}

int main()
{
	int num = getNum();

	printf("%d\n", fib_r(num));
	printf("%d\n", fib_s(num));

	return 0;
}
