Lab 8: Recursive Function
Factorial
  • Basis Step: f(1) = 1
  • Recursive Step: f(n) = n*f(n-1)
    • Pass by Value
    #include <stdio.h>
 
int factorial(int n)
{
	if(n <= 1) // basis step
		return 1;
	else // recursive step
		return n*factorial(n-1);
}
 
int main()
{
	printf("%d\n", factorial(10));
 
	return 0;
}
    Pass by Reference
    #include <stdio.h>
#include <stdlib.h>
 
void factorial(int *p, int n)
{
	if(n <= 1) // basis step
	{
		*p *= 1;
	}
	else // recursive step
	{
		*p *= n;
		factorial(p, n-1);
	}
}
 
int main()
{
	int *ptr;
	ptr = (int *) malloc(sizeof(int));
 
	*ptr = 1;
 
	factorial(ptr, 10);
 
	printf("%d\n", *ptr);
 
	return 0;
}
