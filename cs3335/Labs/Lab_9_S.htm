#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int getNum()
{
	printf("Enter the number of the strings: ");
	int num;
	scanf("%d", &num);

	return num;
}

char *getString()
{
	char *str = (char *) malloc(100*sizeof(char));
	fgets(str, 100, stdin);

	str[strlen(str)-1] = '\0'; // remove \n in the string

	return str;
}

char **getStrings(int num)
{
	char **strings = (char **) malloc(num*sizeof(char *));

	getchar();// read \n character from previous entering

	for(int i = 0; i < num; i++)
	{
		printf("Enter %dth string: ", i+1);
		strings[i] = getString();
	}

	return strings;
}

void display(char **strings, int num)
{
	for(int i = 0; i < num; i++)
		printf("|%s|\n", strings[i]);
}

void sortStrings(char **strings, int size)
{
   for (int i = 0; i < size-1; i++)
       for (int j = 0; j < size-i-1; j++)
           if (strcmp(strings[j], strings[j+1]) > 0)
	   {
		char *s;
		s = strings[j];
		strings[j] = strings[j+1];
		strings[j+1] = s;
	   }
}

int main()
{
	int num = getNum();

	char **strings = getStrings(num);

	sortStrings(strings, num);

	display(strings, num);

	for(int i = 0; i < num; i++)
		free(strings[i]);

	free(strings);

	return 0;
}
