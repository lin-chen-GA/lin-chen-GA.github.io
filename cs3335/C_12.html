Array
Define an Array
#include <stdio.h>
 
void display(int array [], int size)
{
	for(int i = 0; i < size; i++)
		printf("%10d", array[i]);
	printf("\n");
}

int main()
{
	// create array
	int array[10];
	printf("%p\n", array); // address of the first element

	display(array, 10);
 
	// initialize array
	for(int i = 0; i < 10; i++)
		array[i] = i;
	display(array, 10);
 
	return 0;
}

Knowledge Points
  • When an array is created, the default values may not be zero
  • Be able to access a specific element by index
  • index starts from 0
  • array name points to the address of the first element in the array
  • C allows out of boundary access
  • Array can be passed to a function as a regular variable, and the array size is needed
    • Pass Array to Function
  • Array is passed by reference, regular variable is passed by value
    • #include <stdio.h>

void display(int arr [], int size)
{
	for(int i = 0; i < size; i++)
		printf("%10d", arr[i]);
	printf("\n");
}

void change_variable(num)
{
	num *= 2;
	printf("In change_variable function: %d\n", num);
}

void change_array(int arr [], int size)
{
	for(int i = 0; i < size; i++)
		arr[i] *= 2;
	printf("In change_array function: \n");
	display(arr, size);
}

int main()
{
	// pass by value
	int num = 10;

	change_variable(num);

	printf("In main function: %d\n", num);

	// pass by reference
	int array[4];

	for(int i = 0; i < ; i++)
		array[i] = i;

	change_array(array, 4);

	display(array, 4);

	return 0;
}

    Two-Dimensional Array
    #include <stdio.h>

void disp(int array[][4], int rows)
{
	int i, j;
	for(i = 0; i < rows; i++)
	{
		for(j = 0; j < 4; j++)
			printf("%10d", array[i][j]);
		printf("\n");
	}
}

int main()
{
	int array[3][4];

	int i, j;
	for(i = 0; i < 3; i++)
		for(j = 0; j < 4; j++)
			array[i][j] = i*10 + j;

	disp(array, 3);

	return 0;
}
    Array to Pointer
    #include <stdio.h>

int main()
{
	int array[10];

	int i;
	for(i = 0; i < 10; i++)
		array[i] = i;

	int *p;
	p = array;

	for(i = 0; i < 10; i++)
		printf("%d, %d\n", i, *(p+i));

	return 0;
}
