// io.c
#include <stdio.h>
#include "io.h"

int read_num()
{
	int num;
	scanf("%d", &num);

	return num;
}

void display(int size, int array[])
{
	for(int i = 0; i < size; i++)
		printf("%5d", array[i]);
	printf("\n");
}

// util.c
#include <stdlib.h>
#include "util.h"

int get_rand()
{
	return random()%10+1;
}

void assign(int array [], int size)
{
	for(int i = 0; i < size; i++)
	{
		array[i] = get_rand();
	}
}
