#include <stdio.h>
#include <stdlib.h>
 
int main()
{
	FILE * fp; // define a file pointer
	fp = fopen ("info.csv", "w"); // open a file with writing mode
 
	// exit program if the file cannot be openned
	if (fp == NULL)
        	exit(EXIT_FAILURE);
 
	// write into file
	fprintf(fp, "%s,%s\n", "Name", "Lin");
	fprintf(fp, "%s,%s\n", "Office", "2205");
	fprintf(fp, "%s,%s\n", "Hobby", "Swim");
 
	fclose(fp);
 
	return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void processLine(char *str)
{
	char *token;
	token = strtok(str, ",");
	while( token != NULL ) {
		printf( "%s\n", token );
      		token = strtok(NULL, ",");
   	}
}

int main()
{
    FILE* stream = fopen("info.csv", "r");

    char line[1024];

    while (fgets(line, 1024, stream))
    {
	    processLine(line);
    }

    fclose(stream);

    return 0;
}

#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[])
{
	if(argc != 3)
	{
		printf("./sum [num_1] [num_2]\n");
		exit(EXIT_FAILURE);
	}

	int a = atoi(argv[1]);
	int b = atoi(argv[2]);

	printf("%d + %d = %d\n", a, b, a+b);

	return 0;
}
