/// @file util.c
#include <stdio.h>
#include "util.h"

int read_int()
{
	int num;

	printf("Enter an integer number: ");
	scanf("%d", &num);

	return num;
}

int is_even(int n)
{
	if (n % 2 == 0)
		return 1;
	else
		return 0;
}

int get_sum(int a, int b)
{
	return a + b;
}
			
/// @file util.h

/** Read an integer from keyboard
 * @param[out] integer read in
 */
int read_int();

/** Check if an integer is even or odd
 * @param[in] n
 * @param[out] 1/0
 */
int is_even(int n);

/** Calculate the sum of two integers
 * @param[in] a integer number
 * @param[in] b integer number
 * @param[out] sum return summation
 */
int get_sum(int a, int b);
