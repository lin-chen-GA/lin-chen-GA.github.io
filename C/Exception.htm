#include <stdio.h>
#include <errno.h>
#include <string.h>
#include <assert.h>

int main()
{
	int a = 10, b = 0;

	assert (b != 0);

	if(b == 0)
	{
		printf("Errno: %d\n", errno);
		perror("Divided by zero: ");
		fprintf(stderr, "Divided by zero: %s\n", strerror(errno));
	}
	else
		printf("%d / %d = %f\n", a, b, (float)a/b);

	return 0;
}
