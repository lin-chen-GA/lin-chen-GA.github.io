import java.util.Scanner;

public class Ch5_3
{
	public static void main(String [] args)
	{
		Scanner input = new Scanner(System.in);

		boolean cont = true;

		int sum = 0;
		int count = 0;

		while(cont)
		{
			System.out.println("Enter your grade: ");
			int grade = input.nextInt();

			sum += grade;
			count++;

			System.out.println("Are you going to enter more grades (true/false)?");
			cont = input.nextBoolean();
		}

		System.out.println("Average Grade: "+(float)sum/count);
	}
}

    Difference Between While and Do ... While
  • While, the logical expression is evaluated first
  • Do ... While, the logical expression is evaluated after the loop body is executed, the loop body is execuated at least once

    Break and Continue
  • break, terminate the current loop
  • continue, breaks the current iteration, and continues with the next iteration in the loop

    Nested While Loop
    public class Ch5_8
{
	public static void main(String [] args)
	{
		int i = 1;

		while(i <= 9)
		{
			int j = 1;

			while(j <= 9)
			{
				System.out.printf("%d*%d = %2d  ", i, j, i*j);
				j++;
			}

			System.out.println();
			i++;
		}
	}
}

    public class Ch5_11
{
	public static void main(String [] args)
	{
		//define loop index outside for loop
		int i; //scope is main function

		for(i = 0; i < 10; i++)
		{
			System.out.println("Hello World!");
		}

		System.out.println(i); // 10

		//define loop index inside for loop
		for(int j = 0; j < 10; j++) // scope of j is the for loop block
		{
			System.out.println("Hello World!");
		}

		//System.out.println(j); //error
	}
}
