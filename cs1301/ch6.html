    import java.util.Scanner;

public class Ch6_3
{
	public static void main(String [] args)
	{
		Scanner input = new Scanner(System.in);

		while(true)
		{
			System.out.println("Enter an positive integer:");
			int num = input.nextInt();

			System.out.printf("Sum of 1 to %d is: %d\n", num, getSum(num));
		}
	}

	public static int getSum(int n)
	{
		int s = 0;

		for(int i = 1; i <= n; i++)
			s += i;

		return s;
	}
}

	/** Display the first n prime numbers
	 * @param n positive integer number
	 */
	public static void displayPrimes(int n)
	{
		int candidate = 2;
		int count = 0;

		while(count < n)
		{
			if(isPrime(candidate))
			{
				System.out.printf("%5d", candidate);
				count++;
				if(count%5 == 0)

			}

			candidate++;
		}

	}
