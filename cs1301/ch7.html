    import java.util.Scanner;

public class Ch7_4
{
	public static void main(String [] args)
	{
		Scanner input = new Scanner(System.in);

		System.out.print("Enter the array size: ");
		int size = input.nextInt();

		int [] array = new int[size];

		//Initializing arrays with input values
		for(int i = 0; i < array.length; i++)
		{
			System.out.print("Enter the value of the element "+i+": ");
			array[i] = input.nextInt();
		}

		//Displaying array
		for(int i = 0; i < array.length; i++)
			System.out.printf("%10d", array[i]);
		System.out.println();

		//Summing all elements
		int sum = 0;

		for(int i = 0; i < array.length; i++)
			sum += array[i];

		System.out.println("Sum of the array: "+sum);

		//Finding the largest element
		int max = array[0];
		for(int i = 1; i < array.length; i++)
			if(max < array[i]) max = array[i];

		System.out.println("Max of the array: "+max);

		//Find the index of the largest element
		int maxTemp = array[0];
		int index = 0;
		for(int i = 1; i < array.length; i++)
		{
			if(maxTemp < array[i])
			{
				maxTemp = array[i];
				index = i;
			}
		}

		System.out.println("Index of the largest element: "+index);

		//Random Shuffling
		for(int i = 0; i < array.length; i++)
		{
			int j = (int)(Math.random()*array.length);

			//swap element i and element j
			int temp = array[i];
			array[i] = array[j];
			array[j] = temp;
		}

		for(int i = 0; i < array.length; i++)
			System.out.printf("%10d", array[i]);
		System.out.println();

		//Shifting element
		int left = array[0];

		for(int i = 1; i < array.length; i++)
			array[i-1] = array[i];
		array[array.length-1] = left;


		for(int i = 0; i < array.length; i++)
			System.out.printf("%10d", array[i]);
		System.out.println();
	}
}

    Bubble Sort

    public class Ch7_15
{
	public static void bubbleSort(int [] arr)
	{
		for (int i = 0; i < arr.length-1; i++)
	    		for (int j = 0; j < arr.length-i-1; j++)
				if (arr[j] > arr[j+1])
				{
		    			// swap arr[j+1] and arr[i]
					int temp = arr[j];
		    			arr[j] = arr[j+1];
		    			arr[j+1] = temp;
				}
	}

	public static void displayArray(int [] array)
	{
		for(int i = 0; i < array.length; i++)
			System.out.printf("%6d", array[i]);
		System.out.println();
	}

			array[i] = (int)(Math.random()*10)+1; // assign random numbers to array

    Binary Search
  • Search the target in a sorted array
  • Worst case, search log2n times
    • public class Ch7_17
{
	public static boolean binarySearch(int [] array, int target)
	{
		int start = 0, end = array.length-1;

		while(start <= end)
		{
			int middle = (start+end)/2;
			if(array[middle] == target) return true;

			if(target < array[middle])
				end = middle-1;
			else
				start = middle+1;
		}

		return false;
	}

	public static void main(String [] args)
	{
		int [] array = {1, 3, 5, 6, 8, 9, 11, 14, 16, 19, 20};

		System.out.println(binarySearch(array, 8));
		System.out.println(binarySearch(array, 10));
	}
}
