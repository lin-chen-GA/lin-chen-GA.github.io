public class E4_1
{
	public static void main(String [] args)
	{
		System.out.println(Math.sin(Math.toRadians(90))); // sin(90)
		System.out.println(Math.cos(Math.toRadians(90))); // cos(90)
		System.out.println(Math.sin(Math.toRadians(30))); // sin(30)

		System.out.println(Math.sin(Math.PI/2)); // sin(pi/2)
		System.out.println(Math.cos(Math.PI/4)); // cos(pi/4)

		System.out.println(Math.asin(0.5));
		System.out.println(Math.toDegrees(Math.asin(0.5)));
	}
}
