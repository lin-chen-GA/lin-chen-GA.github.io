		y1 = input.nextDouble();

		System.out.println("Enter the coordiante (x2, y2):");
		x2 = input.nextDouble();
		y2 = input.nextDouble();

		System.out.println("Enter the coordiante (x3, y3):");
		x3 = input.nextDouble();
		y3 = input.nextDouble();
 
		//calcualte a, b, c
		a = Math.sqrt((x3-x1)*(x3-x1) + (y3-y1)*(y3-y1));
		b = Math.sqrt((x2-x1)*(x2-x1) + (y2-y1)*(y2-y1));
		c = Math.sqrt((x3-x2)*(x3-x2) + (y3-y2)*(y3-y2));
 
		//calculate the area of the triangle
		double area = a*b/2;
 
		//print out the area
		System.out.println("Area: "+area);

import java.util.Scanner;
 
public class E4_9
{
	public static void main(String [] args)
	{
		Scanner input = new Scanner(System.in);

		//Read user's name
		System.out.print("Please enter user name: ");
		String name = input.nextLine();
 
		//Read user's weight in pound
		System.out.print("User weight (1b):");
		double weight = input.nextDouble();
 
		//Read user's height in inch
		System.out.print("User height (inch):");
		double height = input.nextDouble();
 
		//Calculate BMI
		double BMI = weight/height/height*703;
 
		//Print out user information
		//Name
		//Weight in pound with two significant figures after the decimal point
		//Height in feet and inches
		//BMI with two significant figures after the decimal point
		System.out.printf("%20s:%20s\n", "Name", name);
		System.out.printf("%20s:%20.2f\n", "Weight", weight);
		System.out.printf("%20s:%5d%5s%5.1f%5s\n", "Height", ((int)height)/12, "Feet", height-((int)height)/12*12, "Inch");
		System.out.printf("%20s:%20.2f\n", "BMI", BMI);
	}
}
