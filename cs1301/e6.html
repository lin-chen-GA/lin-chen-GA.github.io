public class Ch6_6
{
	public static int getRandom()
	{
		return 1+(int)(Math.random()*10);
	}

	public static int getSum(int num)
	{
		int s = 0;

		for(int i = 1; i <= num; i++)
			s += i;

		return s;
	}

	public static void printInfo(int s, int n)
	{
		System.out.printf("The sum of 1 to %d is: %d\n", n, s); 
	}

	public static void main(String [] args)
	{
		int num = getRandom();

		int sum = getSum(num);

		printInfo(sum, num);
	}
}

public class Ch6_7
{
	/** Check if a number is prime number
	 * @param num positive integer number
	 * @return true if prime; false otherwise
	 */
	public static boolean isPrime(int num)
	{
		boolean prime = true;

		for(int i = 1; i <= num; i++)
			if(num%i == 0 && i != 1 && i != num)
				return false;

		return true;
	}

	/** Display the first n prime numbers
	 * @param n positive integer number
	 */
	public static void displayPrimes(int n)
	{
		int candidate = 2;
		int count = 0;

		while(count < n)
		{
			if(isPrime(candidate))
			{
				System.out.printf("%5d", candidate);
				count++;
				if(count%5 == 0)
					System.out.println();
			}

			candidate++;
		}

		System.out.println();
	}

	public static void main(String [] args)
	{
		displayPrimes(10);
	}
}

public class E3
{
	/* Convert Celsius Temperature to Fahrenheit Temperature
	 * @param c Celsius Temperature
	 * @return Fahrenheit Temperature
	 */
	public static double c2f(double c)
	{
		return c*9/5+32;
	}

	/* Display a converting table from Celsius temperature to Fahrenheit temperature
	 */
	public static void c2fTable()
	{
		System.out.printf("%20s%20s\n", "Celsius", "Fahrenheit");
		System.out.printf("----------------------------------------\n");
		for(int i = 0; i < 100; i = i+10)
			System.out.printf("%20.1f%20.1f\n", (float)i, c2f(i));
	}

	public static void main(String [] args)
	{
		c2fTable();
	}
}

public class E6
{
	/* Calculate the Greatest Common Divisor
	 * @param a integer number
	 * @param b integer number
	 * @return gcd
	 */
	public static int gcd(int a, int b)
	{
		int gcd = 1;
		int k = 1;

		while(k <= a && k <= b)
		{
			if (a%k == 0 && b%k == 0)
				gcd = k;
			k++;
		}

		return gcd;
	}

	public static void main(String [] args)
	{
		Scanner input = new Scanner(System.in);

		System.out.println("Enter two integer numbers:");
		int a = input.nextInt();
		int b = input.nextInt();

		System.out.println("GCD of %d and %d is: %d\n", a, b, gcd(a, b));
	}
}
