import java.util.Scanner;

public class Change
{
	public static void main(String [] args)
	{
		Scanner input = new Scanner(System.in);

		System.out.printf("%-20s", "Original Amount:");
		double dollars = input.nextDouble();

		int cents = (int)(dollars*100);

		int twenties = cents / 2000;
		cents = cents % 2000;

		int tens = cents / 1000;
		cents = cents % 1000;

		int fives = cents / 500;
		cents = cents % 500;

		int ones = cents / 100;
		cents = cents % 100;

		int quarters = cents / 25;
		cents = cents % 25;

		int dimes = cents / 10;
		cents = cents % 10;

		int nickels = cents / 5;
		
		int pennies = cents % 5;

		System.out.println("****************************************");
		System.out.printf("%-20s%-10d\n", "Twenties:", twenties);
		System.out.printf("%-20s%-10d\n", "Tens:", tens);
		System.out.printf("%-20s%-10d\n", "Fives:", fives);
		System.out.printf("%-20s%-10d\n", "Ones:", ones);
		System.out.printf("%-20s%-10d\n", "Quarters::", quarters);
		System.out.printf("%-20s%-10d\n", "Dimes:", dimes);
		System.out.printf("%-20s%-10d\n", "Nickels:", nickels);
		System.out.printf("%-20s%-10d\n", "Pennies:", pennies);
		System.out.println("*************Have a good day************");
	}
}
