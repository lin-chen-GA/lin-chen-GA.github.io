public int countCode(String str) {
  int count = 0;
  String sub = "code";
  
  for(int i = 0; i <= str.length()-4; i++)
  {
    String temp = str.substring(i, i+4);
    boolean eq = true;
    for(int j = 0; j < sub.length(); j++)
    {
      if (j == 2) continue;
      if(temp.charAt(j) != sub.charAt(j))
      {
        eq = false;
        break;
      }
    }
    if(eq) count++;
  }
  
  return count;
}

public boolean endOther(String a, String b) {
  String aTemp = a.toLowerCase();

  if (aTemp.endsWith(bTemp) || bTemp.endsWith(aTemp))
    return true;
  
  return false;
}

String-2 repeatEnd
public String repeatEnd(String str, int n) {
  String temp = "";
  int length = str.length();
  
  for(int i = 1; i <= n; i++)
    temp = temp + str.substring(length-n, length);
    
  return temp;
}

String-2 repeatSeparator
public String repeatSeparator(String word, String sep, int count) {
  String temp = "";
  
  for(int i = 0; i < count; i++)
    temp = temp + word + sep;
    
  return temp.substring(0, temp.length() - sep.length());
}
String-2 prefixAgain
public boolean prefixAgain(String str, int n) {
  String prefix = str.substring(0, n);
  
  for(int i = 1; i <= str.length()-n; i++)
  {
    if(str.substring(i, i+n).equals(prefix))
      return true;
  }
  
  return false;
}

public boolean xyzMiddle(String str) {
  //find the index of xyz
  for(int i = 0; i <= str.length()-3; i++)
  {
    if(!str.substring(i, i+3).equals("xyz"))
      continue;
      
    int index = i;
  
    int numLeft = index;
    int numRight = str.length() - (index+3);
  
    if(Math.abs(numLeft-numRight) <= 1)
      return true;
  }
  
  return false;
}

public String getSandwich(String str) {
  int start = str.indexOf("bread");
  int end = str.lastIndexOf("bread");
  
  if(start < 0 || end < 0 || start >= end)
    return "";
    
  return str.substring(start+5, end);
}
