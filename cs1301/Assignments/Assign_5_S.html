public int bigDiff(int[] nums) {
  int min = nums[0];
  int max = nums[0];
  
  for(int i = 0; i < nums.length; i++)
    {
      if(nums[i] < min) min = nums[i];
      if(nums[i] > max) max = nums[i];
    }
    
  return max-min;
}

Array-2 shiftLeft
public int[] shiftLeft(int[] nums) {
  int [] array = new int[nums.length];
  
  if(array.length == 0) return array;
  
  for(int i = 1; i < nums.length; i++)
    array[i-1] = nums[i];
    
  array[array.length-1] = nums[0];
  
  return array;
}
Array-2 tenRun
public int[] tenRun(int[] nums) {
  int [] array = new int[nums.length];
  
  boolean start = false;
  int target = 0;
  
  for(int i = 0; i < nums.length; i++)
  {
    if(nums[i] % 10 == 0)
    {
      target = nums[i];
      start = true;
    }
    
    if(start)
      array[i] = target;
    else
      array[i] = nums[i];
  }
  
  return array;
}

Array-3 couintClumps
public int countClumps(int[] nums) {
  if(nums.length == 0) return 0;
  
  int clump = 0;
  boolean isClump = false;
  int current = nums[0];
  
  for(int i = 1; i < nums.length; i++)
  {
    if(nums[i] == current)
    {
      if(!isClump) clump++;
      isClump = true;
    }
    else
    {
      isClump = false;
      current = nums[i];
    }
  }
  
  return clump;
}
