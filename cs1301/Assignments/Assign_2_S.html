public int sumLimit(int a, int b) {
  int sum = a+b;
  
  int d_sum = String.valueOf(sum).length();
  int d_a = String.valueOf(a).length();
  
  if (d_sum > d_a)
    return a;
  else
    return sum;
}

Logic-2 evenlySpaced
public boolean evenlySpaced(int a, int b, int c) {
  if(a > b)
  {
    int temp = a;
    a = b;
    b = temp;
  }
  
  if(b > c)
  {
    int temp = b;
    b = c;
    c = temp;
    if(a >b)
    {
      temp = a;
      a = b;
      b = temp;
    }
  }
  
  if((b-a) == (c-b))
    return true;
  else
    return false;
}
Logic-2 blackjack
public int blackjack(int a, int b) {
  if(a &gt; 21 && b &gt; 21)
    return 0;
    
  if(a &gt; 21) return b;
  if(b &gt; 21) return a;
  
  if((21-a) &lt; (21-b))
    return a;
  else
    return b;
}
