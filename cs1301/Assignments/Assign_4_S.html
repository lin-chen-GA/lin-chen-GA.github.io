Array-1 – commonEnd
public boolean commonEnd(int[] a, int[] b) {
  if(a[0] == b[0]) return true;
  if(a[a.length-1] == b[b.length-1]) return true;
  
  return false;
}
Array-1 – reverse3
public int[] reverse3(int[] nums) {
  int [] array = new int[nums.length];
  array[0] = nums[2];
  array[1] = nums[1];
  array[2] = nums[0];
  
  return array;
}
Array-1 – middleWay
public int[] middleWay(int[] a, int[] b) {
  int [] array = new int[2];
  array[0] = a[1];
  array[1] = b[1];
  
  return array;
}
Array-1 – sum3
public int sum3(int[] nums) {
  int sum = 0;
  
  for(int i = 0; i < nums.length; i++)
    sum += nums[i];
    
  return sum;
}

  int larger = nums[0];

Array-1 – makeEnds
public int[] makeEnds(int[] nums) {
  int [] array = new int[2];
  array[0] = nums[0];
  array[1] = nums[nums.length-1];
  
  return array;
}
Array-1 – rotateLeft3
public int[] rotateLeft3(int[] nums) {
  int [] array = new int[nums.length];
  
  int left = nums[0];
  for(int i = 1; i < nums.length; i++)
    array[i-1] = nums[i];
    
  array[array.length-1] = left;
  
  return array;
}
Array-1 – sum2
public int sum2(int[] nums) {
  if(nums.length == 0) return 0;
  
  if(nums.length == 1) return nums[0];
  
  return nums[0]+nums[1];
}
Array-1 – has23
public boolean has23(int[] nums) {
  for(int i = 0; i < nums.length; i++)
    if(nums[i] == 2 || nums[i] == 3)
      return true;
      
  return false;
}
Array-1 – double23
public boolean double23(int[] nums) {
  int two = 0;
  int three = 0;
  
  for(int i = 0; i < nums.length; i++)
  {
    if(nums[i] == 2) two++;
    if(nums[i] == 3) three++;
  }
  
  if(two == 2 || three == 2)
    return true;
    
  return false;
}
