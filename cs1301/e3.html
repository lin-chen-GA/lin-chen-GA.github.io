			System.out.println("C");
		else if (grade >= 60)
			System.out.println("D");

			System.out.println("F");
	}

import java.util.Scanner;
 
public class E3_6
{
	public static void main(String[] args) {
		Scanner input = new Scanner(System.in);
 
		System.out.println("Enter weight in pounds: ");
		double weight = input.nextDouble();
 
		System.out.println("Enter height in inches: ");
		double height = input.nextDouble();
 
		// convert to standard units, kiograms and meters
		weight = weight * 0.4535; // convert pounds to kiograms
		height = height * 0.0254; // convert inches to meters
 
		// calculate BMI
		double BMI;
		// calculate BMI and print out its value
		BMI = weight/height/height;
 
		// BMI < 18.5, Underweight
		// 18.5 <= BMI < 25.0, Normal
		// 25.0 <= BMI < 30.0, Overweight
		// 30.0 >= BMI, Obese
		// print out BMI Interpretation
		if (BMI < 18.5)
			System.out.println("Underweight");
		else if (BMI < 25)
			System.out.println("Normal");
		else if (BMI < 30)
			System.out.println("Overweight");
		else
			System.out.println("Obese");
	}
}

		int option = input.nextInt();

		System.out.print("Enter the dollar amount: ");
		double amount = input.nextDouble();

		double exchange;

		switch(option)
		{
			case 0: exchange = amount*rate;
			       	System.out.printf("%10.2f dollors is %10.2f RMB ...\n", amount, exchange);
				break;
			case 1: exchange = amount/rate;
			       	System.out.printf("%10.2f RMB is %10.2f dollors ...\n", amount, exchange);
				break;

		}
	}
