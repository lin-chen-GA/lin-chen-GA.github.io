import java.util.Scanner;

public class E1
{
	public static int [][] getArray(int row, int column)
	{
		int [][] array = new int[row][column];

		for(int i = 0; i < row; i++)
			for(int j = 0; j < column; j++)
				array[i][j] = (int)(Math.random()*10) + 1;

		return array;
	}

	public static int [][] transpose(int [][] array)
	{
		int rowT = array[0].length;
		int columnT = array.length;

		int [][] arrayT = new int[rowT][columnT];

		for(int i = 0; i < array.length; i++)
			for(int j = 0; j < array[0].length; j++)
				arrayT[j][i] = array[i][j];

		return arrayT;
	}

	public static int [][] scaleMultiplication(int [][] array, int scale)
	{
		int row = array.length;
		int column = array[0].length;

		int [][] arrayS = new int[row][column];

		for(int i = 0; i < row; i++)
			for(int j = 0; j < column; j++)
				arrayS[i][j] = array[i][j]*scale;

		return arrayS;
	}

	public static int [][] sum(int [][] array_1, int [][] array_2)
	{
		int row = array_1.length;
		int column = array_1[0].length;

		int [][] array = new int[row][column];

		for(int i = 0; i < array.length; i++)
			for(int j = 0; j < array[0].length; j++)
				array[i][j] = array_1[i][j] + array_2[i][j];

		return array;
	}

	public static int rowColumnMul(int [][] array_1, int [][] array_2, int row, int column)
	{
		int sum = 0;
		int size = array_2.length;

		for(int i = 0; i < size; i++)
			sum += array_1[row][i] * array_2[i][column];

		return sum;
	}

	public static int [][] mul(int [][] array_1, int [][] array_2)
	{
		int row = array_1.length;
		int column = array_1[0].length;

		int [][] array = new int[row][column];

		for(int i = 0; i < array.length; i++)
			for(int j = 0; j < array[0].length; j++)
				array[i][j] = rowColumnMul(array_1, array_2, i, j);

		return array;
	}

	public static void display(int [][] array)
	{
		for(int i = 0; i < array.length; i++)
		{
			for(int j = 0; j < array[i].length; j++)
				System.out.printf("%5d", array[i][j]);
			System.out.println();
		}
		System.out.println();
	}

	public static void main(String [] args)
	{
		int [][] array_1 = getArray(3, 3);
		int [][] array_2 = getArray(3, 3);

		display(array_1);
		display(array_2);

		int [][] arrayTranspose = transpose(array_1);
		display(arrayTranspose);

		int [][] arrayScaleMul = scaleMultiplication(array_1, 10);
		display(arrayScaleMul);

		int [][] arraySum = sum(array_1, array_2);
		display(arraySum);

		int [][] arrayMul = mul(array_1, array_2);
		display(arrayMul);
	}
}
