import java.util.Scanner;

public class Lab4
{
	public static void main(String [] args)
	{
		//read number
		Scanner input = new Scanner(System.in);

		System.out.println("Enter the number ");
		int num = input.nextInt();

		if (num < 100 || num > 999) //number is not three-digit number
		{
			System.out.println("Your number "+num+" is not a three-digit number");
		}
		else //number is three-digit number
		{
			int sum = 0;
			sum += Math.pow(num%10, 3);
			sum += Math.pow(num/10%10, 3);
			sum += Math.pow(num/100, 3);

			if (num == sum) //number is Armstrong number
				System.out.println("Your number "+num+" is an Armstrong number");
			else //number is not Armstrong number
				System.out.println("Your number "+num+" is NOT an Armstrong number");
		}
	}
}
