public class Lab13
{
	/** Display array
	 * @param array two-dimensional integer array
	 */
	public static void display(int [][] array)
	{
		for(int i = 0; i < array.length; i++)
		{
			for(int j = 0; j < array[0].length; j++)
			{
				System.out.printf("%5d ", array[i][j]);
			}
			System.out.println();
		}
	}

	/** Generate the required two-dimensional integer array
	 * @param row row number of two-dimensional integer array
	 * @param column column number of two-dimensional integer array
	 * @return two-dimensional integer array
	 */
	public static int [][] init(int row, int column)
	{
		int [][] array = new int[row][column];

		for(int i = 0; i < row; i++)
			for(int j = 0; j < column; j++)
				array[i][j] = i*10+j;

		return array;
	}

	public static void main(String [] args)
	{
		int [][] array = init(5, 10);

		display(array);
	}
}
