public class Lab6
{
	public static void main(String [] args)
	{
		String phrase = "Change is inevitable";
		System.out.println("Length of string: "+phrase.length());

		String mutation1 = phrase + ", except from vending machines";
		System.out.println("Mutation #1: "+mutation1);

		String mutation2 = mutation1.toUpperCase();
		System.out.println("Mutation #2: "+mutation2);

		String mutation3 = mutation2.replace("E", "X");
		System.out.println("Mutation #3: "+mutation3);

		String mutation4 = mutation3.substring(3, 30);
		System.out.println("Mutation #4: "+mutation4);
		System.out.println("Muated length: "+mutation4.length());
	}
}
