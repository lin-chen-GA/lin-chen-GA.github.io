public class Lab9
{
	/** Merge two sorted arrays
	 * @param array_1 sorted integer array
	 * @param array_2 sorted integer array
	 * @return merged sorted integer array
	 */
	public static int [] merge(int [] array_1, int [] array_2)
	{
		int [] array = new int[array_1.length+array_2.length];
		int i = 0; // pointer for array_1
		int j = 0; // pointer for array_2
		int k = 0; // pointer for array

		while(i < array_1.length && j < array_2.length)
		{
			if(array_1[i] < array_2[j])
			{
				array[k] = array_1[i];
				i++;
			}
			else
			{
				array[k] = array_2[j];
				j++;
			}

			k++;
		}

		while(i < array_1.length)
		{
			array[k] = array_1[i];
			k++;
			i++;
		}

		while(j < array_2.length)
		{
			array[k] = array_1[j];
			k++;
			j++;
		}

		return array;
	}

	public static void main(String [] args)
	{
		int [] array1 = {1, 5, 16, 61, 111};
		int [] array2 = {2, 4, 5, 6};

		int [] mergedArray = merge(array1, array2);
		System.out.println(Arrays.toString(mergedArray));
	}
}
			
import java.util.Arrays;

public class Lab_9
{
   /** Merge two sorted arrays
   * @param array_1 sorted integer array
   * @param array_2 sorted integer array
   * @return merged sorted integer array
   */
   public static int [] merge(int [] array1, int [] array2)
   {
   	int [] copy = new int[array1.length + array2.length];
      int left = 0;
      int right = 0;
      
      for(int i = 0; i < copy.length; i++)
      {
         if(left >= array1.length)
         {
            copy[i] = array2[right];
            right++;
            continue;
         }
         
         if(right >= array2.length)
         {
            copy[i] = array1[left];
            left++;
            continue;
         }
         
         if(array1[left] < array2[right])
         {
            copy[i] = array1[left];
            left++;
         }
         else
         {
            copy[i] = array2[right];
            right++;
         }
      }
      
      return copy;
   }
   
   public static void main(String [] args)
   {
   	int [] array1 = {1, 5, 16, 61, 111};
   	int [] array2 = {2, 4, 5, 6};
   
   	int [] mergedArray = merge(array1, array2);
   	System.out.println(Arrays.toString(mergedArray));
   }
}
