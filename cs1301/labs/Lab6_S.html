import java.util.Scanner;

public class Lab7
{
	public static void main(String [] args)
	{
		int num, limit, current;

		Scanner input = new Scanner(System.in);

		System.out.println("Enter an positive integer:");
		num = input.nextInt();

		System.out.println("Enter the upper limit value:");
		limit = input.nextInt();

		current = num;
		int count = 0;
		while(current < limit)
		{
			System.out.printf("%5d ", current);
			count++;
			if (count%5 == 0) System.out.println();
			current += num;
		}
		if(count%5 != 0) System.out.println();
	}
}
