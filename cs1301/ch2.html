public class ComputeArea
{
	public static void main(String [] args)
	{
		//define variables
		double radius;
		double area;

		//assign value to variable
		radius = 20;

		//compute area
		area = radius * radius * 3.1415926;

		//output result
		//string concatenation
		System.out.println("The area for the circle of radius " + radius + " is " + area); 
	}
}

    More Assignment Operators
    public class Assignment
{
	public static void main(String [] args)
	{
		// Augmented Assignment Operators
		int a = 10;

		a += 1; // or a = a + 1
		a -= 1; // or a = a - 1

		a *= 2; // or a = a * 2
		a /= 2; // or a = a / 2
		a %= 2; // or a = a % 2

		// Increment and Decrement Operators
		int b = 100;
		b++; // 101
		b--; // 100

		int c = 100;
		++c; // 101
		--c; // 100

		// What is the difference between i++ and ++i
		int i = 1000;
		System.out.println(i++); // 1000, print out i, then do i = i + 1
		System.out.println(i); // 1001

		int j = 1000;
		System.out.println(++j); // 1001, do j = j + 1, then print out j
		System.out.println(j);
	}
}
    Type Casting
    public class Convert
{
	public static void main(String [] args)
	{
		// implicit type casting
		double a = 10; // convert integer 10 to double 10.0

		System.out.println(9.0/5); // convert integer 5 to double 5.0 since 9.0 is a double number

		// explicit type casting
		double b = 3.14;

		int c = (int)b;

		System.out.println(c);
	}
}
