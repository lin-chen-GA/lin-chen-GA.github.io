		if (num_1 < num_2)
		{
			int temp;
			temp = num_1;
			num_1 = num_2;
			num_2 = temp;
		}

		// print out question
		System.out.println("Question: "+num_1+" - "+num_2+" = ?");

		// student input the answer
		Scanner input = new Scanner(System.in);
		int answer = input.nextInt();

		// check solution
		if ((num_1 - num_2) == answer)
			System.out.println("Correct ...");
		else
			System.out.println("Your Answer is Wrong ...");
	}
}

    import java.util.Scanner;

public class Ch_3_9
{
	public static void main(String[] args) {
		Scanner input = new Scanner(System.in);

		System.out.println("What is your letter score (A|B|C|D|F): ");
		String c = input.next();

		switch (c)
		{
			case "A": System.out.println("You have the score above 90 ...");
					  break; // what is break doing?
			case "B": System.out.println("You have the score above 80 ...");
					  break;
			default:  System.out.println("Work harder ...");
		}
	}
}
    Knowledge Points
  • When the value in a case statement matches the value of the switch-expression, the statements for this case are executed until a break statement or the end of the switch statement
  • The break statement immediately ends the switch statement
  • The default case is optional, which is used to perform actions when one of the specified cases matches the switch-expression
  • Do not forget to use a break statment in a case statement
